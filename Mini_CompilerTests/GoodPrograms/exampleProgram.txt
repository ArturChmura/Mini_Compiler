program
{
   int a, b, t[5,7], i , j;
   read a;
   read b;

   i = 0;
   while(i < 5)
   {
		j = 0;
		while(j < 7)
		{
			t[i,j] = a*i - b|j;
			j = j+1;
			if(t[3,3] < 0)
				break 2;
		}
		i = i+1;
   }

   if(t[3,3] > 0)
   {
	write "positive \n";
   }
   else
   {
	write "not positive \n";
   }

}
